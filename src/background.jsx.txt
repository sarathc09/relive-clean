import { useEffect, useRef } from "react";
import * as THREE from "three";
import GLOBE from "vanta/dist/vanta.globe";

export default function Background() {
  const vantaRef = useRef(null);

  useEffect(() => {
  let effect;

  if (typeof window !== "undefined") {
    effect = GLOBE({
      el: vantaRef.current,
      THREE: THREE,

      mouseControls: true,
      touchControls: true,
      gyroControls: false,

      minHeight: 200,
      minWidth: 200,

      backgroundColor: 0x050816,
      color: 0x10b981,
      color2: 0x22d3ee,

      size: 1.3,
    });
  }

  return () => {
    if (effect) effect.destroy();
  };
}, []);
  return (
    <div
      ref={vantaRef}
      className="absolute inset-0 z-0"
    />
  );
}
